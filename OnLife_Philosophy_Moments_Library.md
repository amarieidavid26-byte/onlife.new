**Body:**

struct PhilosophyMomentView: Identifiable {

### Tracking
```swift
// User defaults or Firebase
var seenMoments: Set<String>
var philosophyMomentsEnabled: Bool
var philosophyMomentsMode: Mode // .on, .subtle, .off
```

### Display Logic
```swift
func shouldShowMoment(_ momentId: String) -> Bool {
    guard philosophyMomentsEnabled else { return false }
    
    switch philosophyMomentsMode {
    case .on:
        return true  // Show icon always, tap to open
    case .subtle:
        return !seenMoments.contains(momentId)  // Only first time
    case .off:
        return false
    }
}
```
